#include <iostream>
using namespace std;

class NumberType
{

private:


    enum enOddOrEven { eOdd = 1, eEven = 2, eZero = 3 };


    static  int ReadNumber()
    {
        int Number;

        cout << "Please enter a number to check if it is odd or even\n";

        cin >> Number;

        return Number;
    }

    static  enOddOrEven CheckNumber()
    {
        int Number = ReadNumber();

        if (Number == 0)
        {
            return eZero;
        }
        else if (Number % 2 == 0)
        {
            return eEven;
        }

        return eOdd;
    }

    public:

    static string PrintNumberType()
    {
        enOddOrEven OddOrEven = CheckNumber();

        switch (OddOrEven)
        {
        case NumberType::eOdd:
            return "Number is odd";
            break;
        case NumberType::eEven:
            return"Number is even";
            break;
        
        default:
            return "Number is zero";
            break;
        }
  
    }

};


int main()
{
    cout << NumberType::PrintNumberType() << endl;
}
