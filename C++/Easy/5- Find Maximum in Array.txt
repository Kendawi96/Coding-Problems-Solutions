#include <iostream>
using namespace std;

int MaxNumberInArray(int Array[], int Arrlength)
{
	int Max = Array[0];

	for (int i = 1 ; i < Arrlength ; i++)
	{
		if (Array[i] > Max)
		{
			Max = Array[i];
		}
	}

	return Max;
}


int main()
{
	int Array[] = { 3,5,2,9,4 };

	int ArrayLength = sizeof(Array) / sizeof(Array[0]);

	cout << "Max number in array = " << MaxNumberInArray(Array, ArrayLength) << endl;
}
