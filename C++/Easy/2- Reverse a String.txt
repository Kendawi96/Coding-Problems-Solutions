#include <iostream>
using namespace std;




string TakeAString()
{
    string Text;

    cout << "Please enter a text or word to Reverse it\n";

    cin >> Text;

    return Text;
}


string ReverseString()
{
    string Text = TakeAString();
    string ReversedText = "";

    for (short i = Text.length() - 1; i >= 0; i--)
    {
        ReversedText += Text[i];
    }

    return ReversedText;
}

int main()
{
    cout << ReverseString() << endl;
}
